### 3.3 Guardian Assignment Logic

Users are assigned a primary guardian based on their saju analysis:

```typescript
interface GuardianAssignment {
  primary: Guardian;      // Strongest element in chart
  secondary: Guardian;    // Second strongest element
  balancing: Guardian;    // Element needed for balance
}

function assignGuardians(elementScores: ElementScores): GuardianAssignment {
  const sorted = Object.entries(elementScores)
    .sort(([,a], [,b]) => b - a);

  const weakest = sorted[sorted.length - 1][0];

  return {
    primary: getGuardianByElement(sorted[0][0]),
    secondary: getGuardianByElement(sorted[1][0]),
    balancing: getGuardianByElement(weakest)
  };
}
```
